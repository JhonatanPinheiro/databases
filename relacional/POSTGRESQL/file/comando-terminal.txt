#Considerando que seu container se chama db_container_postgres e banco/usuário configurados no docker-compose:
#Acessar o banco diretamente dentro do container:
docker exec -it db_container_postgres psql -U admin -d meubanco


#Entrar no psql com usuário apenas e depois escolher o banco:
docker exec -it db_container_postgres psql -U admin


#Depois dentro do prompt do psql:
\c meubanco


#Acessar de fora do container (se o cliente psql estiver instalado no host):
psql -h localhost -p 5432 -U admin -d meubanco

#Se você quiser rodar um comando SQL já na hora de acessar (sem entrar interativamente), pode usar a opção -c. Por exemplo:
docker exec -it db_container_postgres psql -U admin -d meubanco -c "SELECT * FROM tabela_exemplo;"


#Para ver os containers ativos e confirmar o nome:
docker ps

#Para sair do psql, use:
\q

#Se você usar docker-compose (ao invés de nome do container diretamente), pode buscar o ID do container com:
docker-compose ps -q postgres


#Para uma limpeza mais abrangente, removendo containers parados, redes não utilizadas, volumes e imagens não referenciadas:
docker system prune


#Este comando envia um sinal de término (SIGTERM) para todos os containers, permitindo que eles sejam encerrados de forma controlada:
docker stop $(docker ps -q)


|------------ 
⚠️ Atenção
Cuidado ao usar docker kill, pois ele encerra os containers imediatamente, sem permitir que eles realizem um desligamento adequado.
Evite usar docker rm em containers em execução, pois isso pode causar perda de dados ou corrupção, a menos que você tenha certeza de que os containers podem ser removidos sem consequências.
------------|

#Para remover todas as imagens que correspondem a um padrão específico no nome, você pode usar o seguinte comando:
#Por exemplo, para remover todas as imagens que contêm "postgres" no nome:
docker rmi $(docker images --filter=reference='*postgres*' -q)

#Se você conhece o nome exato da imagem (incluindo a tag), pode removê-la diretamente:
docker rmi db_container_postgres:15

#Se a imagem estiver sendo usada por um container, você precisará removê-lo primeiro ou usar a opção -f para forçar a remoção:
docker rmi -f db_container_postgres:15

#Se você deseja remover todas as imagens Docker do seu sistema:
docker rmi $(docker images -q)

#Para remover todos os containers, incluindo os que estão parados, utilize:
docker rm $(docker ps -a -q)

#Se você deseja forçar o encerramento imediato de todos os containers, sem aguardar o processo de término, use:
docker kill $(docker ps -q)



## JOGANDO UM ARQUIVO DE BACKUP VIA TERMINAL USANDO O DOCKER PARA A BASE (LOCALHOST)
	(1) - Coloque o arquivo de backup na pasta do Docker onde fica o arquivo docker-compose
 		Segue caminho: C:\containers_dockers_instalados\postgree


	(2) Depois execute o comando:
 		docker exec -it db_container_postgres bash

	(3) Crie o banco de dados com o nome desejado
		psql -U admin -c "CREATE DATABASE NOME_DO_BANCO_DESEJADO;"

	(4) - Saia do container
 		exit

	(5) - Colocando o arquivo do banco dentro do container
		docker cp NOME_DO_ARQUIVO_BANCO_DE_DADOS db_container_postgres:/tmp/

	(6) - Volte para o container
		docker exec -it db_container_postgres bash

	(7) - Confirma se o arquivo do banco foi jogado para a /tmp/
		ls -lh /tmp/

	(6) Rode o comando para restaurar
		pg_restore --no-owner --no-privileges -U admin -d NOME_BANCO_CRIADO -v /tmp/NOME_DO_ARQUIVO_DO_BANCO_PARA_RESTAURAR.backup


	(8) Verificação de tabelas - Confirme se o banco foi restaurado corretamente:
		psql -U admin -d NOME_DO_BANCO_CRIADO -c "\dt"



## PARA DELETE ALGUM BANCO DE DADOS QUE FOI CRIADO, USANDO DOCKER PELO TERMINAL
	(1 - ETAPA)
		docker exec -it db_container_postgres bash
	
	(2 - ETAPA)
		psql -U admin -d postgres
	
	(5 - ETAPA)
		\l

	(6 - ETAPA (A) )
		DROP DATABASE NOME_DO_BANCO_SER_DELETADO;

	OU
	
	(6 - ETAPA (B) )
		exit

	Fora do container:
		docker exec -it db_container_postgres psql -U admin -d postgres -c "DROP DATABASE nome_do_banco;"
